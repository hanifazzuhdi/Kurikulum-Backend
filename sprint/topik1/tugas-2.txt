/1/ Buat database phpMyAdmin

buat database dengan nama coba_aja

/2/ Buat Tabel

- buat tabel dalam database coba_aja dengan nama pondokit
- buat field dalam tabel

/3/ Buat variabel

$host = "localhost"
$user = "root"
$pass = ""
$database = "coba_aja"
$select = "SELECT * FROM pondokit"

/4/ Buat koneksi 

$conn =  mysqli_connect ($host, $user, $pass, $database);


/5/ Buat fungsi

function namaFungsi ($select)
{
  global $conn;

  // query untuk tampil data
  $result = mysqli_query ($conn, $select)

  // fungsi untuk melooping data 
  $rows = [];
  while ($row = mysqli_fetch_assoc($result)) {
    $rows[] = $row;
  }
  return $rows;
}

/6/ Panggil Fungsi ambil data

$santri = namaFungsi ($conn, $select)

/7/ Tampilkan data

lakukan foreach dari data yang telah disimpan di $santri ke tabel yang telah dibuat di file php
